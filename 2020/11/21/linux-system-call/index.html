Linux系统调用过程

发表于 2020-11-21 | 更新于 2020-11-24 | 分类于 linux

简介

Linux系统调用是如何实现的，参考资料<Linux内核设计与实现>，哈工大操作系统课程。

系统调用

在操作系统中，内核提供了用户进程和内核进行交互的接口。这些接口在应用程序和内核之间扮演了使者的角色，应用程序通过接口向内核发出各种请求，而内核负责满足这些请求。这些接口就是系统调用。系统调用是用户空间访问内核的唯一手段：除异常和陷入外，它们是内核唯一的合法入口。

假设有一个系统调用函数，它只有一个参数。

int func(const char* name);

首先会有_syscallN宏对其处理，_syscallN()是一组宏，其中N的范围从0～6，表示函数参数的个数。

每个宏都有2+2xN个参数，其中第一个参数为函数的返回值，第二个参数为函数的名称，后面2xN个参数依次对应函数中N个参数的类型和名称。

func方法的宏处理如下：

_syscall1(int,func,const char*,name);

_syscall1的定义(i386架构)如下：

#define _syscall1(type, name, type1, arg1)                       
type name(type1 arg1){
    long __res;
    __asm__ volatile(
    "int $0x80"                             
   :"=a"(__res)                           
   :"0"(__NR_##name), "b"((long)(arg1))); 
    __syscall_return(type, __res);                           
}

经过宏替换处理后的func函数如下：

int func(const char* name){
    long __res; 
    __asm__ volatile (
    "int $0x80" 
   :"=a" (__res) //输出
   :"0" (__NR_func),"b" ((long)(name)); //输入
    do{
        if ((unsigned long)(__res) >= (unsigned long)(-127)) { 
            errno = -(__res); 
            __res = -1; 
        } 
        return (int) (__res);
    }while(0);
}

它将系统调用号__NR_func存入到EAX寄存器，系统调用号都定义在unisth.h文件中，然后将参数name存入到EBX寄存器，然后进行int $0x80中断处理。

Linux在初始化过程中会设置中断处理函数，其中int $0x80中断函数设置代码如下：

#define SYSCALL_VECTOR        0x80
set_system_gate(SYSCALL_VECTOR,&system_call);

其中set_system_gate函数定义如下：

static void __init set_system_gate(unsigned int n, void *addr)
{
    _set_gate(idt_table+n,15,3,addr,__KERNEL_CS);
}

其中idt_table用于保存中断描述符，每个中断描述符占用8字节的空间。

struct desc_struct idt_table[256] __attribute__((__section__(".data.idt"))) 
  = { {0, 0}, };
//desc_struct
struct desc_struct {
    unsigned long a,b; //4+4
};

代入参数为：

_set_gate(idt_table+0x80,15,3,&system_call,__KERNEL_CS);

_set_gate是一个宏：

#define _set_gate(gate_addr,type,dpl,addr,seg) 
do { 
    int __d0, __d1; 
    __asm__ __volatile__ (
    "movw %%dx,%%ax\n\t" 
    "movw %4,%%dx\n\t" 
    "movl %%eax,%0\n\t" 
    "movl %%edx,%1" 
   :"=m" (*((long *) (gate_addr))), 
    "=m" (*(1+(long *) (gate_addr))), 
    "=&a" (__d0), 
    "=&d" (__d1) 
   :"i" ((short) (0x8000+(dpl<<13)+(type<<8))), 
    "3" ((char *) (addr)),
    "2" ((seg) << 16)); 
} while (0);

替换参数处理后的代码如下：

do { 
    int __d0, __d1; 
    __asm__ __volatile__ (
    "movw %%dx,%%ax\n\t" //%dx-->%ax
    "movw %4,%%dx\n\t"   //0x8000+(3<<13)+(15<<8)-->%dx
    "movl %%eax,%0\n\t"  //%eax-->idt_table[128].a
    "movl %%edx,%1"      //%edx-->idt_table[128].b
   :"=m" (*((long *) (idt_table+0x80))), 
    "=m" (*(1+(long *) (idt_table+0x80))), 
    "=&a" (__d0),      //__d0 = %eax
    "=&d" (__d1)       //__d1 = %edx
   :"i" ((short) (0x8000+(3<<13)+(15<<8))), 
    "3" ((char *) (&system_call)),
    "2" ((__KERNEL_CS) << 16)); 
} while (0);

上面这段内联汇编代码的主要工作是设置0x80也就是128号中断的描述符。具体过程如下图所示：

应用程序调用系统调用，实际上就是触发0x80中断，该中断会调用system_call方法，system_call根据EAX寄存器中的系统调用号触发相应的系统调用。system_call的方法如下：

ENTRY(system_call)
    pushl %eax    
    SAVE_ALL 
    GET_THREAD_INFO(%ebp)
    cmpl $(nr_syscalls), %eax #判断系统调用号是否合法
    jae syscall_badsys
    testb $_TIF_SYSCALL_TRACE,TI_FLAGS(%ebp)
    jnz syscall_trace_entry
syscall_call:
    call *sys_call_table(,%eax,4) #调用sys_call_table + eax * 4
    movl %eax,EAX(%esp)        
syscall_exit:
    cli            
    movl TI_FLAGS(%ebp), %ecx
    testw $_TIF_ALLWORK_MASK, %cx    
    jne syscall_exit_work

在进入syscall_call之前会先保存用户程序的现场，然后检查系统调用号是否合法，nr_syscalls即系统调用号的数量，由宏定义。

#define nr_syscalls ((syscall_table_size)/4)

系统调用的所有方法都存储在sys_call_table中：

.data
ENTRY(sys_call_table)
    .long sys_restart_syscall
    .long sys_exit
    .long sys_fork
    .long sys_read
    .long sys_write
    .long sys_open        
    ...#省略中间一些系统调用
    .long sys_ni_syscall #无效的系统调用

每个系统调用都和unisth.h中的系统调用号一一对应。

因此应用程序调用系统调用整个过程如下所示：

应用程序调用库函数，库函数通过int 0x80陷入中断，中断处理程序在中断描述符标志中找到128号中断向量，执行system_call方法，system_call通过系统调用号在系统调用表中找到系统调用函数的地址并执行系统调用。

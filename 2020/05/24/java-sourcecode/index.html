ArrayList

数组在创建的时候就需要指定大小，当数组的大小不够用时就需要重新申请更大的内存来存放数据，ArrayList类就被封装为一个可变长的对象数组，能够自动实现扩容。

public boolean add(E e) {
    //确认数组容量是否够,不够就扩容
    ensureCapacityInternal(size + 1);
    elementData[size++] = e;
    return true;
}

private void ensureCapacityInternal(int minCapacity) {
    ensureExplicitCapacity(calculateCapacity(elementData, minCapacity));
}

private static int calculateCapacity(Object[] elementData, int minCapacity) {
    if (elementData == DEFAULTCAPACITY_EMPTY_ELEMENTDATA) {
        return Math.max(DEFAULT_CAPACITY, minCapacity);
    }
    return minCapacity;
}

private void ensureExplicitCapacity(int minCapacity) {
    modCount++;
    //判断minCapacity是否需要扩容
    if (minCapacity - elementData.length > 0)
        grow(minCapacity);
}

private void grow(int minCapacity) {
    int oldCapacity = elementData.length;
    int newCapacity = oldCapacity + (oldCapacity >> 1);//右移1位相当于除以2
    //newCapacity - minCapacity < 0 有两种情况
    //1.oldCapacity=0,此时newCapacity=0
    //2.newCapacity溢出了
    if (newCapacity - minCapacity < 0)
        newCapacity = minCapacity;
    //如果申请的容量超过了Integer.MAX_VALUE - 8,就尝试申请大内存
    if (newCapacity - MAX_ARRAY_SIZE > 0)
        newCapacity = hugeCapacity(minCapacity);
    elementData = Arrays.copyOf(elementData, newCapacity);
}

private static int hugeCapacity(int minCapacity) {
    //minCapacity < 0 即minCapacity已经溢出,抛出异常
    if (minCapacity < 0) 
        throw new OutOfMemoryError();
    return (minCapacity > MAX_ARRAY_SIZE) ?    
        Integer.MAX_VALUE :
        MAX_ARRAY_SIZE;
}

add方法首先去判断elementData的容量是否够用，如果不够用就需要对数组进行扩容，扩容后的大小为原来大小的1.5倍，当申请的1.5倍溢出的时候，就尝试以size + 1的大小申请，如果size + 1还是溢出就会抛出OutOfMemoryError()。

在指定位置添加元素 (0-size)

public void add(int index, E element) {
    //判断插入的index是否有效,区间为[0,size]
    rangeCheckForAdd(index);
    //判断是否需要扩容
    ensureCapacityInternal(size + 1); 
    System.arraycopy(elementData, index, elementData, index + 1,
                     size - index);
    elementData[index] = element;
    size++;
}

private void rangeCheckForAdd(int index) {
    if (index > size || index < 0)
        throw new IndexOutOfBoundsException(outOfBoundsMsg(index));
}

首先判断index在指定的[0,size]范围内，然后再进行扩容判断，随后将index以及之后的元素向后移动一位，将element元素插入到index位置上。

void linkLast(E e) {
    final Node<E> l = last;
    final Node<E> newNode = new Node<>(l, e, null);
    last = newNode;
    //如果链表尾为null,表示链表为空链表
    if (l == null)
        first = newNode;
    else
        l.next = newNode;
    size++;
    modCount++;
}

public void add(int index, E element) {
    //判断插入的index是否有效
    checkPositionIndex(index);    
    //插入到末尾,和add(e)等效
    if (index == size)
        linkLast(element);
    else
        linkBefore(element, node(index));
}

void linkBefore(E e, Node<E> succ) {
    final Node<E> pred = succ.prev;
    final Node<E> newNode = new Node<>(pred, e, succ);
    succ.prev = newNode;
    if (pred == null)
        first = newNode;
    else
        pred.next = newNode;
    size++;
    modCount++;
}

Node<E> node(int index) {
    //选择遍历方式,index在[0,size/2]时,从头向尾遍历,否则从尾向头遍历
    if (index < (size >> 1)) {
        Node<E> x = first;
        for (int i = 0; i < index; i++)
            x = x.next;
        return x;
    } else {
        Node<E> x = last;
        for (int i = size - 1; i > index; i--)
            x = x.prev;
        return x;
    }
}

由于LinkedList是双向链表，因此可以根据index的位置选择从头向后遍历还是从尾向前遍历。

public E remove(int index) {
    //判断删除的index是否有效
    checkElementIndex(index);

}

E unlink(Node<E> x) {
    final E element = x.item;
    final Node<E> next = x.next;
    final Node<E> prev = x.prev;
        //当前结点为头结点
    if (prev == null) {
        first = next;
    } else {
        prev.next = next;
        x.prev = null;
    }
        //当前结点为尾结点
    if (next == null) {
        last = prev;
    } else {
        next.prev = prev;
        x.next = null;
    }

    x.item = null;
    size--;
    modCount++;
    return element;
}

public HashMap(int initialCapacity, float loadFactor) {
    if (initialCapacity < 0)
        throw new IllegalArgumentException("Illegal initial capacity: " +
                                           initialCapacity);
    if (initialCapacity > MAXIMUM_CAPACITY)
        initialCapacity = MAXIMUM_CAPACITY;
    if (loadFactor <= 0 || Float.isNaN(loadFactor))
        throw new IllegalArgumentException("Illegal load factor: " +
                                           loadFactor);
    this.loadFactor = loadFactor;
    //寻找一个2^n,使得 2^(n-1) < initialCapacity <= 2^n
    this.threshold = tableSizeFor(initialCapacity);
}

static final int tableSizeFor(int cap) {
    //先减1,确保原本就是2^n不会变成2^(n+1)
    int n = cap - 1;
    //原理为找到最高位的1,然后使后面的都为1即可,int只有32位
    //最高位向右移动了一位
    n |= n >>> 1;
    n |= n >>> 2;
    n |= n >>> 4;
    n |= n >>> 8;
    n |= n >>> 16;
    return (n < 0) ? 1 : (n >= MAXIMUM_CAPACITY) ? MAXIMUM_CAPACITY : n + 1;
}

public HashMap(Map<? extends K, ? extends V> m) {

final V putVal(int hash, K key, V value, boolean onlyIfAbsent,
               boolean evict) {
    Node<K,V>[] tab; Node<K,V> p; int n, i;
    //如果table没有初始化或者长度为0,就resize()
    if ((tab = table) == null || (n = tab.length) == 0)
        n = (tab = resize()).length;
    //通过hash值查找到链表,如果当前链表为空,就生成一个新Node
    if ((p = tab[i = (n - 1) & hash]) == null)
        tab[i] = newNode(hash, key, value, null);
    else {
        //否则判断插入的key是否已经存在
        Node<K,V> e; K k;
        //如果链表头就是要找的元素
        if (p.hash == hash &&
            ((k = p.key) == key || (key != null && key.equals(k))))
            e = p;
        //如果链表已经转换为红黑树
        else if (p instanceof TreeNode)
            //插入元素到红黑树
            e = ((TreeNode<K,V>)p).putTreeVal(this, tab, hash, key, value);
        else {
            //遍历链表
            for (int binCount = 0; ; ++binCount) {
                //链表中不存在put的元素
                if ((e = p.next) == null) {
                    //插入元素
                    p.next = newNode(hash, key, value, null);
                    //链表长度超过8
                    if (binCount >= TREEIFY_THRESHOLD - 1) // -1 for 1st
                        //尝试转红黑树
                        treeifyBin(tab, hash);
                    break;
                }
                //链表中存在put的元素
                if (e.hash == hash &&
                    ((k = e.key) == key || (key != null && key.equals(k))))
                    break;
                p = e;
            }
        }
        //e!=null,即存在put的元素,只需要修改元素的value即可
        if (e != null) {
            V oldValue = e.value;
            if (!onlyIfAbsent || oldValue == null)
                e.value = value;
            afterNodeAccess(e);
            return oldValue;
        }
    }
    ++modCount;
    //插入后的size如果超过threshold,resize()
    //先size=size+1
      //再比较size和threshold
    if (++size > threshold)
        resize();
    afterNodeInsertion(evict);
    return null;
}

final Node<K,V>[] resize() {
    Node<K,V>[] oldTab = table;
    int oldCap = (oldTab == null) ? 0 : oldTab.length;
    int oldThr = threshold;
    int newCap, newThr = 0;
    //oldCap>0,table已经存在
    if (oldCap > 0) {
        if (oldCap >= MAXIMUM_CAPACITY) {
            threshold = Integer.MAX_VALUE;
            return oldTab;
        }
        else if ((newCap = oldCap << 1) < MAXIMUM_CAPACITY &&
                 oldCap >= DEFAULT_INITIAL_CAPACITY)
            newThr = oldThr << 1;
    }
    //oldCap=0,table不存在
    //1.oldThr>0,调用的是自定义的容量的构造函数
    else if (oldThr > 0) 
        newCap = oldThr;
    //2. oldThr=0,调用默认的构造函数
    else {               
        newCap = DEFAULT_INITIAL_CAPACITY; //newCap=16
        newThr = (int)(DEFAULT_LOAD_FACTOR * DEFAULT_INITIAL_CAPACITY);//newThr=12
    }
    //重新计算threshold
    if (newThr == 0) {
        float ft = (float)newCap * loadFactor;
        newThr = (newCap < MAXIMUM_CAPACITY && ft < (float)MAXIMUM_CAPACITY ?
                  (int)ft : Integer.MAX_VALUE);
    }
    threshold = newThr;
    @SuppressWarnings({"rawtypes","unchecked"})
    //以新的容量重新申请一个数组
    Node<K,V>[] newTab = (Node<K,V>[])new Node[newCap];
    table = newTab;
    //重新分配元素到新数组上
    if (oldTab != null) {
        for (int j = 0; j < oldCap; ++j) {
            Node<K,V> e;
            if ((e = oldTab[j]) != null) {
                oldTab[j] = null;
                if (e.next == null)
                    newTab[e.hash & (newCap - 1)] = e;
                else if (e instanceof TreeNode)
                    ((TreeNode<K,V>)e).split(this, newTab, j, oldCap);
                else {
                    Node<K,V> loHead = null, loTail = null;
                    Node<K,V> hiHead = null, hiTail = null;
                    Node<K,V> next;
                    do {
                        next = e.next;
                        if ((e.hash & oldCap) == 0) {
                            if (loTail == null)
                                loHead = e;
                            else
                                loTail.next = e;
                            loTail = e;
                        }
                        else {
                            if (hiTail == null)
                                hiHead = e;
                            else
                                hiTail.next = e;
                            hiTail = e;
                        }
                    } while ((e = next) != null);
                    if (loTail != null) {
                        loTail.next = null;
                        newTab[j] = loHead;
                    }
                    if (hiTail != null) {
                        hiTail.next = null;
                        newTab[j + oldCap] = hiHead;
                    }
                }
            }
        }
    }
    return newTab;
}

final void treeifyBin(Node<K,V>[] tab, int hash) {
    int n, index; Node<K,V> e;
    //如果数组长度没有达到64,调用resize()
    if (tab == null || (n = tab.length) < MIN_TREEIFY_CAPACITY)
        resize();
    //链表转二叉树
    else if ((e = tab[index = (n - 1) & hash]) != null) {
        TreeNode<K,V> hd = null, tl = null;
        do {
            TreeNode<K,V> p = replacementTreeNode(e, null);
            if (tl == null)
                hd = p;
            else {
                p.prev = tl;
                tl.next = p;
            }
            tl = p;
        } while ((e = e.next) != null);
        if ((tab[index] = hd) != null)
            hd.treeify(tab);
    }
}

    Node<K,V>[] tab; Node<K,V> first, e; int n; K k;

            return first;
        if ((e = first.next) != null) {
            if (first instanceof TreeNode)
                return ((TreeNode<K,V>)first).getTreeNode(hash, key);
            do {
                if (e.hash == hash &&
                    ((k = e.key) == key || (key != null && key.equals(k))))
                    return e;

    }

final Node<K,V> removeNode(int hash, Object key, Object value,
                           boolean matchValue, boolean movable) {
    Node<K,V>[] tab; Node<K,V> p; int n, index;
    if ((tab = table) != null && (n = tab.length) > 0 &&
        (p = tab[index = (n - 1) & hash]) != null) {
        Node<K,V> node = null, e; K k; V v;
        if (p.hash == hash &&
            ((k = p.key) == key || (key != null && key.equals(k))))
            node = p;
        else if ((e = p.next) != null) {
            if (p instanceof TreeNode)
                node = ((TreeNode<K,V>)p).getTreeNode(hash, key);
            else {
                do {
                    if (e.hash == hash &&
                        ((k = e.key) == key ||
                         (key != null && key.equals(k)))) {
                        node = e;
                        break;
                    }
                    p = e;
                } while ((e = e.next) != null);
            }
        }
        if (node != null && (!matchValue || (v = node.value) == value ||
                             (value != null && value.equals(v)))) {
            if (node instanceof TreeNode)
                ((TreeNode<K,V>)node).removeTreeNode(this, tab, movable);
            else if (node == p)
                tab[index] = node.next;
            else
                p.next = node.next;
            ++modCount;
            --size;
            afterNodeRemoval(node);
            return node;
        }
    }
    return null;
}

//方法使用synchronized,所以是线程安全的,而HashMap是线程不安全的
public synchronized V put(K key, V value) {
    //不允许value为null,而HashMap是允许的
    if (value == null) {
        throw new NullPointerException();
    }
    Entry<?,?> tab[] = table;
    //调用key.hashCode(),所以key也不允许为null
    int hash = key.hashCode();
    // HashMap的key计算方法 key.hashCode()^(key.hashCode() >>> 16)
    // Hatable的key计算方法 key.hashCode()&0x7FFFFFFF(hashCode可能为负数,&0x7FFFFFFF强制转正数)
    int index = (hash & 0x7FFFFFFF) % tab.length;
    @SuppressWarnings("unchecked")
    Entry<K,V> entry = (Entry<K,V>)tab[index];
    //遍历链表
    for(; entry != null ; entry = entry.next) {
        //修改元素
        if ((entry.hash == hash) && entry.key.equals(key)) {
            V old = entry.value;
            entry.value = value;
            return old;
        }
    }
    //添加元素
    addEntry(hash, key, value, index);
    return null;
}

private void addEntry(int hash, K key, V value, int index) {
    modCount++;
    Entry<?,?> tab[] = table;
    //扩容时机和HashMap不同
    //HashTable先扩容再插入数据
    //HashMap先插入数据再扩容
    if (count >= threshold) {
        //扩容
        rehash();
        tab = table;
        hash = key.hashCode();
        index = (hash & 0x7FFFFFFF) % tab.length;
    }
    @SuppressWarnings("unchecked")
    Entry<K,V> e = (Entry<K,V>) tab[index];
    tab[index] = new Entry<>(hash, key, value, e);
    count++;
}

protected void rehash() {
    int oldCapacity = table.length;
    Entry<?,?>[] oldMap = table;
    //容量翻倍加1
    int newCapacity = (oldCapacity << 1) + 1;
    //扩容后需要的容量[Integer.MAX_VALUE-7,Integer.MAX_VALUE]之间
    if (newCapacity - MAX_ARRAY_SIZE > 0) {
        //若已经扩容到上限了,不再扩容
        if (oldCapacity == MAX_ARRAY_SIZE)
            return;
        //扩容到Integer.MAX_VALUE-8
        newCapacity = MAX_ARRAY_SIZE;
    }
    //申请新数组
    Entry<?,?>[] newMap = new Entry<?,?>[newCapacity];
    modCount++;
    threshold = (int)Math.min(newCapacity * loadFactor, MAX_ARRAY_SIZE + 1);
    table = newMap;
    //重新分配元素到数组上
    for (int i = oldCapacity ; i-- > 0 ;) {
        //头插法
        for (Entry<K,V> old = (Entry<K,V>)oldMap[i] ; old != null ; ) {
            Entry<K,V> e = old;
            old = old.next;
            int index = (e.hash & 0x7FFFFFFF) % newCapacity;
            e.next = (Entry<K,V>)newMap[index];
            newMap[index] = e;
        }
    }
}
  1. get
public synchronized V get(Object key) {
    Entry<?,?> tab[] = table;
    int hash = key.hashCode();
    int index = (hash & 0x7FFFFFFF) % tab.length;
    for (Entry<?,?> e = tab[index] ; e != null ; e = e.next) {
        if ((e.hash == hash) && e.key.equals(key)) {
            return (V)e.value;
        }
    }
    return null;
}
  1. remove
public synchronized V remove(Object key) {
    Entry<?,?> tab[] = table;
    int hash = key.hashCode();
    int index = (hash & 0x7FFFFFFF) % tab.length;
    @SuppressWarnings("unchecked")
    Entry<K,V> e = (Entry<K,V>)tab[index];
    for(Entry<K,V> prev = null ; e != null ; prev = e, e = e.next) {
        if ((e.hash == hash) && e.key.equals(key)) {
            modCount++;
            if (prev != null) {
                prev.next = e.next;
            } else {
                tab[index] = e.next;
            }
            count--;
            V oldValue = e.value;
            e.value = null;
            return oldValue;
        }
    }
    return null;
}

public final void acquire(int arg) {
    //1.tryAcquire尝试去获得锁,为抽象方法,由具体子类来实现
    //2.tryAcquire失败时,在等待队列末尾新增一个Node
    //3.acquireQueued尝试从在等待队列中获得锁
    //4.如果当前线程获得锁失败且被中断,当前线程中断
    //如果当前线程获得锁,就不会被中断
    if (!tryAcquire(arg) &&
        acquireQueued(addWaiter(Node.EXCLUSIVE), arg))
        selfInterrupt();
}
//addWaiter
private Node addWaiter(Node mode) {
    Node node = new Node(Thread.currentThread(), mode);
    Node pred = tail;
    //链表尾不为null,此时链表不为空链表
    if (pred != null) {
        //尾插法
        node.prev = pred;
        //CAS
        if (compareAndSetTail(pred, node)) {
            pred.next = node;
            return node;
        }
    }
    //链表为空或CAS失败
    enq(node);
    return node;
}
//enq
private Node enq(final Node node) {
    //死循环,一直尝试
    for (;;) {
        Node t = tail;
        //链表为空的情况
        if (t == null) {
            if (compareAndSetHead(new Node()))
                tail = head;
        } else {
            node.prev = t;
            if (compareAndSetTail(t, node)) {
                t.next = node;
                return t;
            }
        }
    }
}
//acquireQueued
final boolean acquireQueued(final Node node, int arg) {
    boolean failed = true;
    try {
        boolean interrupted = false;
        //死循环,一直尝试
        for (;;) {
            final Node p = node.predecessor();
            //先驱结点为第一个结点时,尝试去获取锁成功(初始化时head为一个空的Node结点)
            if (p == head && tryAcquire(arg)) {
                //设置当前结点为头结点,同时剔除p结点
                setHead(node);
                //p.next = node
                p.next = null;
                failed = false;
                return interrupted;
            }
            //先判断结点线程是否应该被挂起,并在需要时挂起线程
            if (shouldParkAfterFailedAcquire(p, node) &&
                parkAndCheckInterrupt())
                interrupted = true;
        }
    } finally {
        if (failed)
            cancelAcquire(node);
    }
}
//shouldParkAfterFailedAcquire
private static boolean shouldParkAfterFailedAcquire(Node pred, Node node) {
    int ws = pred.waitStatus;//获取前一个结点的状态
    if (ws == Node.SIGNAL)
        return true;
    //static final int CANCELLED =  1;
    //ws大于0即表示该结点已经不请求获取锁
    if (ws > 0) {
        do {
            //从链表中剔除状态为Node.CANCELLED的结点
            node.prev = pred = pred.prev;
        } while (pred.waitStatus > 0);
        pred.next = node;
    } else {
        //尝试将前一个结点的状态改为Node.SIGNAL
        compareAndSetWaitStatus(pred, ws, Node.SIGNAL);
    }
    return false;
}
//parkAndCheckInterrupt
private final boolean parkAndCheckInterrupt() {
    //挂起线程
    LockSupport.park(this);
    //被唤醒后,返回线程是否被中断(该方法会清除中断标识)
    return Thread.interrupted();
}
  1. release
public final boolean release(int arg) {
    //tryRelease尝试释放锁,为抽象方法,由具体子类来实现 
    if (tryRelease(arg)) {
        Node h = head;
        //头结点不为空,且头结点waitStatus不为0
        if (h != null && h.waitStatus != 0)
            //唤醒阻塞的头结点线程
            unparkSuccessor(h);
        return true;
    }
    return false;
}
//unparkSuccessor
private void unparkSuccessor(Node node) {
    int ws = node.waitStatus;
    if (ws < 0)
        compareAndSetWaitStatus(node, ws, 0);
    Node s = node.next;
    //后续结点不存在或者后续结点已经被取消
    if (s == null || s.waitStatus > 0) {
        s = null;
        //从后向前找一个有效结点
        for (Node t = tail; t != null && t != node; t = t.prev)
            if (t.waitStatus <= 0)
                s = t;
    }
    if (s != null)
        //唤醒结点线程
        LockSupport.unpark(s.thread);
}

  1. await
public final void await() throws InterruptedException {
    //判断当前线程是否被中断
    if (Thread.interrupted())
        throw new InterruptedException();
    //在条件队列链表尾增加一个等待结点
    Node node = addConditionWaiter();
    //释放所有的重入锁
    int savedState = fullyRelease(node);
    int interruptMode = 0;
    while (!isOnSyncQueue(node)) {
        //挂起线程
        LockSupport.park(this);
        //唤醒后判断线程是否被中断,非中断跳出while
        if ((interruptMode = checkInterruptWhileWaiting(node)) != 0)
            break;
    }
    //尝试去获得锁
    if (acquireQueued(node, savedState) && interruptMode != THROW_IE)
        interruptMode = REINTERRUPT;
    if (node.nextWaiter != null) // clean up if cancelled
        unlinkCancelledWaiters();
    if (interruptMode != 0)
        reportInterruptAfterWait(interruptMode);
}
//addConditionWaiter
private Node addConditionWaiter() {
    Node t = lastWaiter;
    if (t != null && t.waitStatus != Node.CONDITION) {
        //删除被取消的结点
        unlinkCancelledWaiters();
        t = lastWaiter;
    }
    Node node = new Node(Thread.currentThread(), Node.CONDITION);
    //单向链表
    //这里不用考虑并发的原因是调用await的线程必然获得了排他锁,因此不可能同时有2个线程进入这段区域
    if (t == null)
        firstWaiter = node;
    else
        t.nextWaiter = node;
    lastWaiter = node;
    return node;
}
//fullyRelease
final int fullyRelease(Node node) {
    boolean failed = true;
    try {
        int savedState = getState();
        if (release(savedState)) {
            failed = false;
            return savedState;
        } else {
            throw new IllegalMonitorStateException();
        }
    } finally {
        if (failed)
            node.waitStatus = Node.CANCELLED;
    }
}
  1. signalAll
public final void signalAll() {
    //没有加锁的情况下调用signalAll方法
    if (!isHeldExclusively())
        throw new IllegalMonitorStateException();
    Node first = firstWaiter;
    if (first != null)
        doSignalAll(first);
}
//doSignalAll
private void doSignalAll(Node first) {
    lastWaiter = firstWaiter = null;
    do {
        Node next = first.nextWaiter;
        first.nextWaiter = null;
        transferForSignal(first);
        first = next;
    } while (first != null);
}
//transferForSignal
final boolean transferForSignal(Node node) {
    //结点已经被取消,就跳过该结点
    if (!compareAndSetWaitStatus(node, Node.CONDITION, 0))
        return false;
    //将结点添加到锁等待队列中,返回值为前一个结点
    Node p = enq(node);
    int ws = p.waitStatus;
    //前一个结点处于取消状态,或者CAS失败,唤醒线程
    if (ws > 0 || !compareAndSetWaitStatus(p, ws, Node.SIGNAL))
        LockSupport.unpark(node.thread);
    return true;
}
  1. signal
public final void signal() {
    //没有加锁的情况下调用signal方法
    if (!isHeldExclusively())
        throw new IllegalMonitorStateException();
    Node first = firstWaiter;
    if (first != null)
        doSignal(first);
}
//doSignal
private void doSignal(Node first) {
    do {
        //first.nextWaiter==null说明仅仅只有一个结点,将lastWaiter也赋值为null
        if ( (firstWaiter = first.nextWaiter) == null)
            lastWaiter = null;
        //去除掉关系
        first.nextWaiter = null;
    } while (!transferForSignal(first) &&
             (first = firstWaiter) != null);
}

//继承自Sync
final boolean nonfairTryAcquire(int acquires) {
    final Thread current = Thread.currentThread();
    int c = getState();
    //当前锁没有竞争者
    if (c == 0) {
        //CAS
        if (compareAndSetState(0, acquires)) {
            setExclusiveOwnerThread(current);
            return true;
        }
    }
    //锁的拥有者为当前线程,重入
    else if (current == getExclusiveOwnerThread()) {
        //累计重入次数
        int nextc = c + acquires;
        //重入次数溢出
        if (nextc < 0)
            throw new Error("Maximum lock count exceeded");
        //设置新的state,即重入次数
        setState(nextc);
        return true;
    }
    return false;
}
//FairSync
protected final boolean tryAcquire(int acquires) {
    final Thread current = Thread.currentThread();
    int c = getState();
    if (c == 0) {
        //唯一的不同就是多了一步hasQueuedPredecessors判断,这一步判断链表是否为空,或者当前线程为链表头
        //表现为非抢占式
        if (!hasQueuedPredecessors() &&
            compareAndSetState(0, acquires)) {
            setExclusiveOwnerThread(current);
            return true;
        }
    }
    else if (current == getExclusiveOwnerThread()) {
        int nextc = c + acquires;
        if (nextc < 0)
            throw new Error("Maximum lock count exceeded");
        setState(nextc);
        return true;
    }
    return false;
}

//总结:
//线程数小于corePoolSize,直接创建线程
//线程数等于corePoolSize,添加到等待队列
//若等待队列满,若线程数小于maximumPoolSize,新建线程
//若等待队列满,若线程数等于maximumPoolSize,说明线程池已经无法添加任务,执行拒绝策略
public void execute(Runnable command) {
    //command不能为null
    if (command == null)
        throw new NullPointerException();
    //获取线程池状态
    int c = ctl.get();
    //1.当前线程数量小于corePoolSize
    if (workerCountOf(c) < corePoolSize) {
        //调用addWorker,从corePoolSize创建线程
        if (addWorker(command, true))
            return;
        //如果添加失败,重新获取线程池状态
        c = ctl.get();
    }
    //2.当前线程数已经大于corePoolSize
    //只有线程状态为RUNNING时才接受新任务,尝试向等待队列中添加任务
    if (isRunning(c) && workQueue.offer(command)) {
        int recheck = ctl.get();
        //线程池状态发生变化,尝试从等待队列中删除任务
        if (! isRunning(recheck) && remove(command))
            reject(command);
        //两种情况1.是RUNNING状态,2.不是RUNNING状态,但是从等待队列中移除任务失败
        //如果当前活动线程是为0,就新建一个线程
        else if (workerCountOf(recheck) == 0)
            addWorker(null, false);
    }
    //3.等待队列已满,调用addWorker,从maximumPoolSize中创建线程
    else if (!addWorker(command, false))
        //还是无法创建,调用拒绝策略
        reject(command);
}
//addWorker方法
//参数
//firstTask:任务
//core:true从corePoolSize创建,false从maximumPoolSize创建
//返回值
//true新建线程并启动成功,false线程启动失败
private boolean addWorker(Runnable firstTask, boolean core) {
    retry:
    //死循环
    for (;;) {
        //获取运行线程池状态
        int c = ctl.get();
        int rs = runStateOf(c);
        //1.线程池状态不为RUNNING,不接受新任务
        //2.存在一种特殊情况,状态为SHUTDOWN,不接受新任务，但是需要执行已经提交的任务
        if (rs >= SHUTDOWN &&
            ! (rs == SHUTDOWN &&
               firstTask == null &&
               ! workQueue.isEmpty()))
            return false;
                //线程状态为RUNNING
        for (;;) {
            //获取当前任务数量
            int wc = workerCountOf(c);
            //1.线程数量超出上限2^29-1
            //2.线程数量超过corePoolSize或者maximumPoolSize
            //不接受新任务
            if (wc >= CAPACITY ||
                wc >= (core ? corePoolSize : maximumPoolSize))
                return false;
            //递增c,若成功跳出外层for循环
            if (compareAndIncrementWorkerCount(c))
                break retry;
            //获取线程状态
            c = ctl.get();
            //如果线程状态发生了变更,从外层循环重新开始,否则继续尝试加1
            if (runStateOf(c) != rs)
                continue retry;
        }
    }
    //任务是否启动成功
    boolean workerStarted = false;
    //任务是否添加成功
    boolean workerAdded = false;
    Worker w = null;
    try {
        //新建一个Worker
        w = new Worker(firstTask);
        final Thread t = w.thread;
        if (t != null) {
            final ReentrantLock mainLock = this.mainLock;
            mainLock.lock();
            try {
                int rs = runStateOf(ctl.get());
                //1.线程池处于RUNNING状态
                //2.线程池处于SHUTDOWN,但是
                if (rs < SHUTDOWN ||
                    (rs == SHUTDOWN && firstTask == null)) {
                    //如果线程已经被启动,抛出异常,防止重复启动同一个线程
                    if (t.isAlive()) 
                        throw new IllegalThreadStateException();
                    //在HashSet中添加
                    workers.add(w);
                    int s = workers.size();
                    //更新线程池中最大的线程数
                    if (s > largestPoolSize)
                        largestPoolSize = s;
                    workerAdded = true;
                }
            } finally {
                mainLock.unlock();
            }
            //如果任务已经被添加到HashSet中,启动线程
            if (workerAdded) {
                t.start();
                workerStarted = true;
            }
        }
    } finally {
        //如果没有启动成功
        if (! workerStarted)
            //回滚
            addWorkerFailed(w);
    }
    return workerStarted;
}

ThreadPoolExecutor的任务是委托给内部类Worker来执行的，具体由Work来控制。

private final class Worker extends AbstractQueuedSynchronizer implements Runnable{
    final Thread thread;//执行线程
    Runnable firstTask;//初始任务
    volatile long completedTasks;//线程完成任务数
}

Worker(Runnable firstTask) {
    setState(-1);
    this.firstTask = firstTask;
    this.thread = getThreadFactory().newThread(this);
}

//Worker
public void run() {
    runWorker(this);
}
//ThreadPoolExecutor
final void runWorker(Worker w) {
    Thread wt = Thread.currentThread();
    //获取任务
    Runnable task = w.firstTask;
    w.firstTask = null;
    w.unlock();
    boolean completedAbruptly = true;
    try {
        //当前task不为null,否则从等待队列中获取任务
        while (task != null || (task = getTask()) != null) {
            //线程上锁
            w.lock();
            //
            if ((runStateAtLeast(ctl.get(), STOP) ||
                 (Thread.interrupted() &&
                  runStateAtLeast(ctl.get(), STOP))) &&
                !wt.isInterrupted())
                wt.interrupt();
            try {
                //执行前操作,空方法,由子类实现
                beforeExecute(wt, task);
                Throwable thrown = null;
                try {
                    //调用task的run,不再另外启动线程
                    task.run();
                } catch (RuntimeException x) {
                    thrown = x; throw x;
                } catch (Error x) {
                    thrown = x; throw x;
                } catch (Throwable x) {
                    thrown = x; throw new Error(x);
                } finally {
                    ////执行后操作,空方法,由子类实现
                    afterExecute(task, thrown);
                }
            } finally {
                task = null;
                //线程完成任务数量自增
                w.completedTasks++;
                //线程解锁
                w.unlock();
            }
        }
        completedAbruptly = false;
    } finally {
        processWorkerExit(w, completedAbruptly);
    }
}
